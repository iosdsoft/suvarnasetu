    <script>

        document.addEventListener("DOMContentLoaded", function() {

            var userAgent = navigator.userAgent || navigator.vendor || window.opera;


            // URLs to the App Store and Google Play Store

            var iosUrl = "https://apps.apple.com/app/id6499589328"; // iOS App Store URL

            var androidUrl = "https://play.google.com/store/apps/details?id=com.dsoft.kvirchandbhaigoldpalace"; // Android Google Play Store URL


            // Check if the device is iOS

            if (/iPad|iPhone|iPod/.test(userAgent) && !window.MSStream) {

                window.location.href = iosUrl;

            }

            // Check if the device is Android

            else if (/android/i.test(userAgent)) {

                window.location.href = androidUrl;

            }

            // Fallback for other devices or unsupported platforms

            else {

                window.location.href = androidUrl"; // Optional: a fallback URL for unsupported platforms

            }

        });
