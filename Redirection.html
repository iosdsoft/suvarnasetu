<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Redirecting...</title>

    <script>

        document.addEventListener("DOMContentLoaded", function() {

            var userAgent = navigator.userAgent || navigator.vendor || window.opera;


            // URLs to the App Store and Google Play Store

            var iosUrl = "https://apps.apple.com/us/app/your-app-name/idXXXXXXXXX"; // Replace with your app's iOS App Store URL

            var androidUrl = "https://play.google.com/store/apps/details?id=com.yourcompany.yourapp"; // Replace with your app's Google Play Store URL


            // Check if the device is iOS

            if (/iPad|iPhone|iPod/.test(userAgent) && !window.MSStream) {

                window.location.href = iosUrl;

            } 

            // Check if the device is Android

            else if (/android/i.test(userAgent)) {

                window.location.href = androidUrl;

            } 

            // Fallback for other devices or unsupported platforms

            else {

                alert("Please visit this page on an iOS or Android device to download the app.");

            }

        });

    </script>

</head>

<body>

</body>

</html>
